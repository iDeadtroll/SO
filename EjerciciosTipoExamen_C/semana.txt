#include <stdio.h>

int main() {
    // Declarar el array de los días de la semana
    char* diasSemana[] = {"Lunes", "Martes", "Miércoles", "Jueves", "Viernes", "Sábado", "Domingo"};

    // Imprimir los días de la semana
    for (int i = 0; i < 7; i++) {
        printf("%s\n", diasSemana[i]);
    }

    return 0;
}
